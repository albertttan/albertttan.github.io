def log_func(x, a, b):
    return a * np.log(1 + b * x)

def sigmoid_adjusted(x):  # 0 <= x < \infty, 0 <= f(x) < 1
    return 2 / (1 + np.exp(-x)) - 1

class Module(): 
    def __init__(self): 
        self.age = 0  # minutes
        self.mass = 125000  # grams adsorbent
        self.current_contamination = 0.1  # miligrams pollutant / gram adsorbent
        self.input_concentration = 8.5  # miligrams pollutant / liter
    def calculate_contamination_inverse(self):  # inverse function of log_fit()
        return (np.exp(self.current_contamination / param_a) - 1) / param_b_adjusted
    def calculate_rate(self):  # derivative of log_fit() & impact of input concentration
        return no_stirring_coefficient * (self.input_concentration / standard_input_concentration) * (param_a * param_b_adjusted / (1 + param_b_adjusted * self.calculate_contamination_inverse()))
    def simulation(self): 
        self.current_absorption_rate = self.calculate_rate()  # miligrams pollutant / gram adsorbent * minute
        self.current_raw_absorbed_mass = self.current_absorption_rate * self.mass  # miligrams pollutant / minute
        self.current_raw_absorbed_concentration = self.current_raw_absorbed_mass / water_flow  # miligrams pollutant / liter
        self.current_adjustedusted_absorbed_concentration = self.input_concentration * sigmoid_adjusted(self.current_raw_absorbed_concentration / self.input_concentration)  # miligrams pollutant / liter
        self.age += 1  # minutes
        self.current_contamination += self.current_adjustedusted_absorbed_concentration * water_flow / self.mass   # miligrams pollutant / gram adsorbent
        return (self.input_concentration - (self.current_adjustedusted_absorbed_concentration))  # miligrams pollutant / liter

modules = [Module(), Module(), Module()]
output_concentration = []

for time in range(129600):  # 90 days period
    if modules[0].age >= replacement_threshold: 
        modules.pop(0)
        modules.append(Module())
    current_concentration = 0.5
    for module in modules: 
        module.input_concentration = current_concentration
        current_concentration = module.simulation()
    output_concentration.append(current_concentration)
